Níveis de iluminamento com leds







Código C++


int valor = 0;

void setup()
{
pinMode(A0, INPUT);
Serial.begin(9600);
pinMode(2, OUTPUT);
pinMode(3, OUTPUT);
pinMode(4, OUTPUT);
}

void loop()
{
valor = analogRead(A0);
Serial.println(valor);
if (valor >= 300) {
digitalWrite(2, HIGH);
digitalWrite(3, LOW);
digitalWrite(4, LOW);
}
if (valor >= 600) {
digitalWrite(3, HIGH);
digitalWrite(2, LOW);
digitalWrite(4, LOW);
}
if (valor >= 900) {
digitalWrite(4, HIGH);
digitalWrite(3, LOW);
digitalWrite(2, LOW);
}
delay(100);
}
